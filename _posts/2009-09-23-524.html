stage.addEventListener(Event.RESIZE, resizeHandler);
stage.align = StageAlign.TOP_LEFT;
stage.scaleMode = StageScaleMode.NO_SCALE;

var map:Map = new Map();

map.key = "YOUR-GOOGLE-MAP-API-KEY";

map.setSize(new Point(stage.stageWidth, stage.stageHeight));
map.addEventListener(MapEvent.MAP_READY, onMapReady);
this.addChild(map);
Logger.putVardump(map.key, "test_data");

map.setCenter(new LatLng(35.667129,139.749613), 17, MapType.SATELLITE_MAP_TYPE);

new MarkerOptions({
strokeStyle: new StrokeStyle({color: 0x333333}),
fillStyle: new FillStyle({color: 0xFFFFFF, alpha: 1}),
radius:10,
hasShadow:true
})
);
