<?php
$username = "ユーザー名";
$password = "パスワード";
$host = "大体は「localhost」";
$database = "データベース名";
$mysql = mysql_connect($host,$username,$password);
if(mysql_select_db($database, $mysql)) {
print "OK";
}else {
print "NG";
}
/*データを表示する例*/
$result = mysql_query('SQL文', $mysql);
while ($data = mysql_fetch_array($result)) {
echo $data['フィールド名など'] . "<br>\n";
}
?>
